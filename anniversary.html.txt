    <script>
        // Create floating hearts
        function createHeart() {
            const heart = document.createElement('div');
            heart.className = 'heart';
            heart.innerHTML = '💖';
            heart.style.left = Math.random() * 100 + '%';
            heart.style.animationDelay = Math.random() * 2 + 's';
            heart.style.fontSize = (Math.random() * 10 + 15) + 'px';
            document.getElementById('hearts').appendChild(heart);

            setTimeout(() => {
                heart.remove();
            }, 8000);
        }

        // Create hearts periodically
        setInterval(createHeart, 800);

        // Animate timeline on scroll
        function animateTimeline() {
            const timelineItems = document.querySelectorAll('.timeline-item');
            
            timelineItems.forEach(item => {
                const itemTop = item.getBoundingClientRect().top;
                const windowHeight = window.innerHeight;
                
                if (itemTop < windowHeight * 0.8) {
                    item.classList.add('animate');
                }
            });
        }

        // Envelope opening functionality
        function openEnvelope() {
            const envelope = document.querySelector('.envelope');
            const messageContent = document.getElementById('messageContent');
            
            envelope.classList.add('open');
            
            setTimeout(() => {
                messageContent.classList.add('show');
                typeMessage();
            }, 800);
        }

        // Typewriter effect
        function typeMessage() {
            const message = "Dear Mom & Dad,\n\nThank you for showing me what true love looks like. Your journey together inspires me every day. Watching you both support each other through life's ups and downs has taught me the real meaning of commitment and partnership.\n\nYour love story is my favorite fairy tale - one that gets more beautiful with each passing year. Here's to many more years of laughter, adventures, and endless love!\n\nHappy 19th Anniversary!\n\nWith all my love,\nYour Child 💕";
            
            const typewriter = document.getElementById('typewriter');
            let i = 0;
            
            function typeChar() {
                if (i < message.length) {
                    if (message.charAt(i) === '\n') {
                        typewriter.innerHTML += '<br>';
                    } else {
                        typewriter.innerHTML += message.charAt(i);
                    }
                    i++;
                    setTimeout(typeChar, 50);
                }
            }
            
            typeChar();
        }

        // Music controls
        let musicPlaying = false;
        const music = document.getElementById('bgMusic');
        
        function toggleMusic() {
            const btn = document.getElementById('musicBtn');
            
            if (musicPlaying) {
                music.pause();
                btn.innerHTML = '🎵';
                musicPlaying = false;
            } else {
                music.play().catch(() => {
                    // Handle autoplay restrictions
                    console.log('Music autoplay blocked by browser');
                });
                btn.innerHTML = '⏸️';
                musicPlaying = true;
            }
        }

        // Scroll animations
        window.addEventListener('scroll', () => {
            animateTimeline();
        });

        // Initialize animations on load
        window.addEventListener('load', () => {
            animateTimeline();
        });

        // Smooth scrolling for better UX
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });
    </script>
